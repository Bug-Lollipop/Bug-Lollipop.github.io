const path = require('path')

const HtmlWebpackPlugin = require('html-webpack-plugin')
const webpack = require('webpack')

const moduleName = process.env.MODULE_NAME || 'ares'

const modulePath = path.join(__dirname, './src', moduleName)

const moduleConfig = {
goblin: {
outputDir: 'dist/goblin',
entry: './src/goblin/main.js',
},
ares: {
outputDir: 'dist/ares',
entry: './src/goblin/main.js',
},
}

const DEFINE_PARAM = {
VERSION: Date.now(),
NODE_ENV: JSON.stringify(process.env.NODE_ENV),
MODULE_NAME: JSON.stringify(process.env.MODULE_NAME),
}

module.exports = {
publicPath: process.env.NODE_ENV === 'production' ? '__FAAS_CDN__' : '/',
outputDir: moduleConfig[moduleName].outputDir,
configureWebpack: {
entry: moduleConfig[moduleName].entry,
resolve: {
alias: {
'@': path.resolve('./src'),
'@goblin': path.resolve('./src/goblin'),
'@ares': path.resolve('./src/ares'),
'&': path.resolve('./src/common'),
},
},
plugins: [
new HtmlWebpackPlugin({
filename: 'index.html',
template: path.join(modulePath, 'index.html'),
templateParameters: DEFINE_PARAM,
}),
new webpack.DefinePlugin(DEFINE_PARAM),
],
},
devServer: {
proxy: {
'/api': {
target: 'http://finance.alta.elenet.me',
changeOrigin: true,
},
},
},
}
